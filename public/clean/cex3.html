Clean & Clear Code

예시3 : 복합범위함수


Bad Code

 
 
            let x = 7, y = 5; 
            
            const addAndSub = (a,b) => {
                 // 덧셈 
                const addResult = a+b; 
                // 뺄셈 
                const subResult = a-b; 
                // 결과값 
                return `A=${a}, B=${b} >>> 덧셈결과=${addResult} / 뺄셈결과=${subResult}`; 
            }

Good Code

 

            let resultString = "", x = 7, y = 5; 

            const doAdd = (a,b) => {
                return a+b; 
            }
    
            const doSub = (a,b) => {
                return a-b; 
            }
    
            resultString =  `A=${x}, B=${y} >>> 덧셈결과=${doAdd(x,y)} / 뺄셈결과=${doSub(x,y)}`;
